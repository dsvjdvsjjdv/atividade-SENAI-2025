#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd (0x27, 16, 2);

 #define Botao1 0
 #define Botao2 145
 #define Botao3 329
 #define Botao4 505
 #define Botao5 741



void setup()
{
  Serial.begin(9600);
  
 
   lcd.init();
   lcd.backlight();
  
}
void loop()
{
  int valor = analogRead(A0);
  if (valor == Botao1)
  {
   lcd.println("Botao1 apertado ");
  }
  else if(valor == Botao2)
  {
    lcd.setCursor(0,0);
    lcd.println("Botao2 apertado");
   
  }
  else if(valor == Botao3) 
  {
    lcd.setCursor(0,0);    
    lcd.println("Botao3 apertado");
   
  }
  else if(valor == Botao4) 
  {
    lcd.setCursor(0,0);
    lcd.println("Botao4 apertado");
  
  }
  else if(valor == Botao4) 
  {
    lcd.setCursor(0,0);
    lcd.println("Botao5 apertado");
    
  }
  
}
